25def compute_portfolio_resonance(session: Session) -> list[dict]: 

26 """ 

27 計算所有大師的最新持倉與使用者關注清單／實際持倉的重疊。 

28 

29 流程： 

30 1. 載入所有啟用中大師的最新持倉（tickers） 

31 2. 載入使用者的 Stock 關注清單 tickers 

32 3. 載入使用者的 Holding 實際持倉 tickers 

33 4. 合併使用者 tickers，對每位大師呼叫 compute_resonance_matches() 

34 5. 回傳帶有 overlapping_tickers 與各股票最新動作的列表 

35 

36 Args: 

37 session: Database session 

38 

39 Returns: 

40 list of dicts，每筆代表一位大師的共鳴結果： 

41 guru_id, guru_display_name, 

42 overlapping_tickers (list[str]), 

43 overlap_count (int), 

44 holdings (list[dict]: ticker, action, weight_pct, change_pct) 

45 """ 

46 gurus = find_all_active_gurus(session) 

47 if not gurus: 

48 logger.info("無啟用中大師，跳過共鳴計算。") 

49 return [] 

50 

51 # 使用者關注清單（Stock 表，is_active=True） 

52 watchlist_tickers: set[str] = { 

53 s.ticker 

54 for s in session.exec( 

55 select(Stock).where(Stock.is_active == True) # noqa: E712 

56 ).all() 

57 } 

58 

59 # 使用者實際持倉（Holding 表） 

60 holding_tickers: set[str] = {h.ticker for h in session.exec(select(Holding)).all()} 

61 

62 user_tickers = watchlist_tickers | holding_tickers 

63 

64 results: list[dict] = [] 

65 for guru in gurus: 

66 holdings = find_holdings_by_guru_latest(session, guru.id) 

67 guru_ticker_map: dict[str, dict] = {} 

68 for h in holdings: 

69 if h.ticker: 

70 guru_ticker_map[h.ticker] = { 

71 "ticker": h.ticker, 

72 "action": h.action, 

73 "weight_pct": h.weight_pct, 

74 "change_pct": h.change_pct, 

75 } 

76 

77 guru_tickers = set(guru_ticker_map.keys()) 

78 overlapping = compute_resonance_matches(guru_tickers, user_tickers) 

79 

80 results.append( 

81 { 

82 "guru_id": guru.id, 

83 "guru_display_name": guru.display_name, 

84 "overlapping_tickers": sorted(overlapping), 

85 "overlap_count": len(overlapping), 

86 "holdings": [guru_ticker_map[t] for t in sorted(overlapping)], 

87 } 

88 ) 

89 

90 # Sort by overlap_count descending so most relevant gurus appear first 

91 results.sort(key=lambda x: x["overlap_count"], reverse=True) 

92 logger.info( 

93 "共鳴計算完成，%d 位大師，共有 %d 位有重疊持倉。", 

94 len(results), 

95 sum(1 for r in results if r["overlap_count"] > 0), 

96 ) 

97 return results

100def get_resonance_for_ticker(session: Session, ticker: str) -> list[dict]: 

101 """ 

102 查詢哪些大師的最新申報中持有指定股票（用於 Radar 頁面徽章顯示）。 

103 

104 Pre-loads all active gurus and their latest filings in bulk to avoid N+1 queries. 

105 

106 Args: 

107 session: Database session 

108 ticker: 股票代號 

109 

110 Returns: 

111 list of dicts，每筆代表一位持有此股票的大師： 

112 guru_id, guru_display_name, 

113 action, weight_pct, change_pct, 

114 report_date, filing_date 

115 """ 

116 holdings = find_holdings_by_ticker_across_gurus(session, ticker) 

117 if not holdings: 

118 return [] 

119 

120 # Pre-load all active gurus and latest filings to avoid N+1 queries 

121 all_gurus = find_all_active_gurus(session) 

122 guru_map: dict[int, str] = {g.id: g.display_name for g in all_gurus} 

123 filing_map: dict[int, object] = { 

124 g.id: find_latest_filing_by_guru(session, g.id) for g in all_gurus 

125 } 

126 

127 results: list[dict] = [] 

128 for holding in holdings: 

129 filing = filing_map.get(holding.guru_id) 

130 results.append( 

131 { 

132 "guru_id": holding.guru_id, 

133 "guru_display_name": guru_map.get(holding.guru_id, ""), 

134 "action": holding.action, 

135 "weight_pct": holding.weight_pct, 

136 "change_pct": holding.change_pct, 

137 "report_date": filing.report_date if filing else None, # type: ignore[union-attr] 

138 "filing_date": filing.filing_date if filing else None, # type: ignore[union-attr] 

139 } 

140 ) 

141 

142 return results

145def get_great_minds_list(session: Session) -> list[dict]: 

146 """ 

147 回傳「英雄所見略同」清單：使用者（關注清單或持倉）＋至少一位大師同時持有的股票。 

148 

149 按持有該股票的大師數量降序排列（大師共識越高越靠前）。 

150 

151 Args: 

152 session: Database session 

153 

154 Returns: 

155 list of dicts，每筆代表一支共鳴股票： 

156 ticker, 

157 guru_count (int), 

158 gurus (list[dict]: guru_id, guru_display_name, action, weight_pct) 

159 """ 

160 resonance_results = compute_portfolio_resonance(session) 

161 if not resonance_results: 

162 return [] 

163 

164 # Aggregate: ticker → list of guru dicts 

165 ticker_gurus: dict[str, list[dict]] = {} 

166 for entry in resonance_results: 

167 for holding in entry["holdings"]: 

168 tk = holding["ticker"] 

169 if tk not in ticker_gurus: 

170 ticker_gurus[tk] = [] 

171 ticker_gurus[tk].append( 

172 { 

173 "guru_id": entry["guru_id"], 

174 "guru_display_name": entry["guru_display_name"], 

175 "action": holding["action"], 

176 "weight_pct": holding["weight_pct"], 

177 } 

178 ) 

179 

180 great_minds = [ 

181 { 

182 "ticker": ticker, 

183 "guru_count": len(gurus), 

184 "gurus": gurus, 

185 } 

186 for ticker, gurus in ticker_gurus.items() 

187 ] 

188 great_minds.sort(key=lambda x: x["guru_count"], reverse=True) 

189 return great_minds
