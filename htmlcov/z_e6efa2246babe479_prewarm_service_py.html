57def prewarm_all_caches() -> None: 

58 """非阻塞式啟動預熱 — 填充 L1/L2 快取。 

59 

60 流程： 

61 1. 技術訊號（batch_download_history + prime_signals_cache_batch，一次 HTTP 請求） 

62 2. 其餘各階段並行執行：moat、fear_greed、etf_holdings、beta、 

63 guru_backfill、sector、etf_sector_weights 

64 

65 整個流程以 try/except 包裹，確保任何失敗都不會影響應用程式正常運作。 

66 """ 

67 start = time.monotonic() 

68 logger.info("快取預熱啟動...") 

69 

70 try: 

71 tickers = _collect_tickers() 

72 except Exception as exc: 

73 logger.error("快取預熱：無法讀取資料庫，中止預熱。%s", exc, exc_info=True) 

74 _set_prewarm_ready(True) # 服務仍可運作，只是快取為冷啟動狀態 

75 return 

76 

77 if not tickers["all"]: 

78 logger.info("快取預熱：資料庫中無任何股票或持倉，跳過預熱。") 

79 _set_prewarm_ready(True) 

80 return 

81 

82 logger.info( 

83 "快取預熱：共 %d 檔標的（signals=%d, moat=%d, sector=%d, etf=%d, beta=%d, etf_sector_weights=%d）", 

84 len(tickers["all"]), 

85 len(tickers["signals"]), 

86 len(tickers["moat"]), 

87 len(tickers["sector"]), 

88 len(tickers["etf"]), 

89 len(tickers["beta"]), 

90 len(tickers["etf"]), 

91 ) 

92 

93 # Phase 1: 技術訊號（批次下載，大幅減少 HTTP 請求數量） 

94 _prewarm_phase("signals", lambda: _batch_prewarm_signals(tickers["signals"])) 

95 

96 # Phase 2+: 其餘階段並行執行（互相獨立，不依賴 Phase 1 結果） 

97 # moat 預熱使用 4 個工作執行緒（預設 SCAN_THREAD_POOL_SIZE=2 限制吞吐量； 

98 # 提高並行度讓執行緒在全域限流器釋放時立即接手，縮短整體等待時間） 

99 _MOAT_PREWARM_WORKERS = 4 

100 parallel_phases: list[tuple[str, object]] = [ 

101 ("fear_greed", get_fear_greed_index), 

102 ( 

103 "moat", 

104 lambda: prewarm_moat_batch( 

105 tickers["moat"], max_workers=_MOAT_PREWARM_WORKERS 

106 ), 

107 ), 

108 ("guru_backfill", _backfill_all_gurus), 

109 ] 

110 if tickers["etf"]: 

111 parallel_phases.append( 

112 ("etf_holdings", lambda: prewarm_etf_holdings_batch(tickers["etf"])) 

113 ) 

114 parallel_phases.append( 

115 ("etf_sector_weights", lambda: _prewarm_etf_sector_weights(tickers["etf"])) 

116 ) 

117 if tickers["beta"]: 

118 parallel_phases.append(("beta", lambda: prewarm_beta_batch(tickers["beta"]))) 

119 if tickers["sector"]: 

120 parallel_phases.append(("sector", lambda: _prewarm_sectors(tickers["sector"]))) 

121 

122 with ThreadPoolExecutor(max_workers=min(len(parallel_phases), 4)) as pool: 

123 phase_futures = { 

124 pool.submit(_prewarm_phase, name, fn): name for name, fn in parallel_phases 

125 } 

126 for future in as_completed(phase_futures): 

127 phase_name = phase_futures[future] 

128 try: 

129 future.result() 

130 except Exception as exc: 

131 logger.warning( 

132 "快取預熱並行階段 [%s] 發生未預期錯誤：%s", phase_name, exc 

133 ) 

134 

135 elapsed = time.monotonic() - start 

136 logger.info("快取預熱完成，耗時 %.1f 秒。", elapsed) 

137 _set_prewarm_ready(True)

145def _collect_tickers() -> dict[str, list[str]]: 

146 """從 DB 收集需要預熱的 ticker 清單。 

147 

148 回傳 dict: 

149 - all: 所有 unique tickers（watchlist + holdings，排除 Cash 類） 

150 - signals: 需要技術訊號的 tickers（排除 Cash） 

151 - moat: 需要護城河分析的 tickers（排除 Bond/Cash） 

152 - etf: 需要 ETF 成分股的 tickers（is_etf=True） 

153 - beta: 需要 Beta 值的 tickers（壓力測試用，排除 Cash） 

154 """ 

155 with Session(engine) as session: 

156 # 活躍追蹤清單 

157 stocks = list( 

158 session.exec(select(Stock).where(Stock.is_active == True)).all() # noqa: E712 

159 ) 

160 # 持倉（排除現金） 

161 holdings = list( 

162 session.exec( 

163 select(Holding).where( 

164 Holding.user_id == DEFAULT_USER_ID, 

165 Holding.is_cash == False, # noqa: E712 

166 ) 

167 ).all() 

168 ) 

169 

170 stock_map = {s.ticker: s for s in stocks} 

171 holding_tickers = {h.ticker for h in holdings} 

172 

173 # Union of watchlist + holdings (unique) 

174 all_tickers = set(stock_map.keys()) | holding_tickers 

175 

176 # Signals: 排除 Cash 類 

177 signals_tickers = [ 

178 t 

179 for t in all_tickers 

180 if t not in stock_map 

181 or stock_map[t].category.value not in SKIP_SIGNALS_CATEGORIES 

182 ] 

183 

184 # Moat: 排除 Bond/Cash 及 ETF（ETF 無損益表，moat 分析必定失敗） 

185 moat_tickers = [ 

186 t 

187 for t in all_tickers 

188 if t not in stock_map 

189 or ( 

190 stock_map[t].category.value not in SKIP_MOAT_CATEGORIES 

191 and not stock_map[t].is_etf 

192 ) 

193 ] 

194 

195 # ETF: 只有追蹤清單中 is_etf=True 的標的 

196 etf_tickers = [t for t, s in stock_map.items() if s.is_etf] 

197 

198 # Beta: 與 signals 使用相同範圍（排除 Cash） 

199 # 若未來需要不同過濾邏輯（如包含 Bond），再拆分 

200 beta_tickers = signals_tickers 

201 

202 # Equity: 與 rebalance 端點 sector_exposure 使用相同範圍（EQUITY_CATEGORIES） 

203 # 確保預熱涵蓋所有需要板塊資訊的標的，避免首次請求顯示 "Unknown" 

204 equity_tickers = [ 

205 t 

206 for t in all_tickers 

207 if t not in stock_map or stock_map[t].category.value in EQUITY_CATEGORIES 

208 ] 

209 

210 # Sector: 熱力圖需要包含 Bond 類標的（Bond 有 GICS 板塊，只是不做訊號/護城河） 

211 # SKIP_SIGNALS_CATEGORIES（僅 Cash）排除的標的，其餘均需板塊資訊 

212 sector_tickers = [ 

213 t 

214 for t in all_tickers 

215 if t not in stock_map 

216 or stock_map[t].category.value not in SKIP_SIGNALS_CATEGORIES 

217 ] 

218 

219 return { 

220 "all": sorted(all_tickers), 

221 "signals": sorted(signals_tickers), 

222 "moat": sorted(moat_tickers), 

223 "etf": sorted(etf_tickers), 

224 "beta": sorted(beta_tickers), 

225 "equity": sorted(equity_tickers), 

226 "sector": sorted(sector_tickers), 

227 }

230def _batch_prewarm_signals(signal_tickers: list[str]) -> None: 

231 """批次下載所有標的歷史資料（一次 HTTP 請求），再並行計算訊號填充快取。 

232 若批次下載失敗或部分標的缺資料，回退至逐一呼叫。 

233 """ 

234 if not signal_tickers: 

235 return 

236 

237 hist_batch = batch_download_history(signal_tickers) 

238 if hist_batch: 

239 primed = prime_signals_cache_batch(hist_batch) 

240 logger.info( 

241 "快取預熱 [signals] 批次預熱 %d/%d 檔。", primed, len(signal_tickers) 

242 ) 

243 # 回退：批次下載中缺失的標的（非 US 市場、資料不足等） 

244 missed = [t for t in signal_tickers if t not in hist_batch] 

245 if missed: 

246 logger.info( 

247 "快取預熱 [signals] 回退至個別呼叫：%d 檔（%s）。", 

248 len(missed), 

249 ", ".join(missed), 

250 ) 

251 prewarm_signals_batch(missed) 

252 else: 

253 # 批次下載完全失敗，回退至個別呼叫 

254 logger.warning("快取預熱 [signals] 批次下載失敗，回退至個別呼叫。") 

255 prewarm_signals_batch(signal_tickers)

297def _prewarm_sectors(tickers: list[str]) -> None: 

298 """對股票類持倉並行呼叫 get_ticker_sector()，填充磁碟快取。 

299 

300 以 ThreadPoolExecutor 並行處理；失敗的單筆記錄警告後繼續，不中斷整個預熱流程。 

301 """ 

302 total = len(tickers) 

303 ok_count = 0 

304 

305 def _fetch_one(ticker: str) -> tuple[str, bool]: 

306 sector = get_ticker_sector(ticker) 

307 logger.debug("快取預熱 [sector] %s → %s", ticker, sector or "N/A") 

308 return ticker, True 

309 

310 with ThreadPoolExecutor(max_workers=SCAN_THREAD_POOL_SIZE) as pool: 

311 futures = {pool.submit(_fetch_one, t): t for t in tickers} 

312 for future in as_completed(futures): 

313 ticker = futures[future] 

314 try: 

315 future.result() 

316 ok_count += 1 

317 except Exception as exc: 

318 logger.warning("快取預熱 [sector] %s 失敗：%s", ticker, exc) 

319 logger.info("快取預熱 [sector] 完成 %d/%d 筆。", ok_count, total)

348def _prewarm_phase(name: str, fn) -> None: 

349 """執行單一預熱階段，失敗時記錄警告但不中斷後續階段。""" 

350 try: 

351 phase_start = time.monotonic() 

352 fn() 

353 elapsed = time.monotonic() - phase_start 

354 logger.info("快取預熱 [%s] 完成，耗時 %.1f 秒。", name, elapsed) 

355 except Exception as exc: 

356 logger.warning("快取預熱 [%s] 失敗（非致命）：%s", name, exc, exc_info=True)
