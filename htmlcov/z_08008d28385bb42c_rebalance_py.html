14def calculate_rebalance( 

15 category_values: dict[str, float], 

16 target_config: dict[str, float], 

17 threshold: float = DRIFT_THRESHOLD_PCT, 

18) -> dict: 

19 """ 

20 計算再平衡分析。 

21 

22 Args: 

23 category_values: 各分類的實際市值 {"Bond": 50000.0, "Cash": 20000.0, ...} 

24 target_config: 目標配置百分比 {"Bond": 50, "Cash": 10, ...} 

25 threshold: 偏移門檻（百分點），超過此值才產生建議 

26 

27 Returns: 

28 { 

29 "total_value": float, 

30 "categories": {cat: {"target_pct", "current_pct", "drift_pct", "market_value"}}, 

31 "advice": [str, ...], 

32 } 

33 """ 

34 total_value = sum(category_values.values()) 

35 if total_value <= 0: 

36 return { 

37 "total_value": 0.0, 

38 "categories": {}, 

39 "advice": [{"key": "rebalance.advice_zero", "params": {}}], 

40 } 

41 

42 all_categories = sorted( 

43 set(list(target_config.keys()) + list(category_values.keys())) 

44 ) 

45 categories_result: dict[str, dict] = {} 

46 advice: list[str] = [] 

47 

48 for cat in all_categories: 

49 target_pct = target_config.get(cat, 0.0) 

50 mv = category_values.get(cat, 0.0) 

51 current_pct = round((mv / total_value) * 100, 2) 

52 drift = round(current_pct - target_pct, 2) 

53 

54 categories_result[cat] = { 

55 "target_pct": target_pct, 

56 "current_pct": current_pct, 

57 "drift_pct": drift, 

58 "market_value": round(mv, 2), 

59 } 

60 

61 if abs(drift) > threshold: 

62 icon = CATEGORY_ICON.get(cat, "📊") 

63 if drift > 0: 

64 advice.append( 

65 { 

66 "key": "rebalance.advice_over", 

67 "params": { 

68 "icon": icon, 

69 "category": cat, 

70 "drift": f"{drift:.1f}", 

71 }, 

72 } 

73 ) 

74 else: 

75 advice.append( 

76 { 

77 "key": "rebalance.advice_under", 

78 "params": { 

79 "icon": icon, 

80 "category": cat, 

81 "drift": f"{abs(drift):.1f}", 

82 }, 

83 } 

84 ) 

85 

86 if not advice: 

87 advice.append({"key": "rebalance.advice_ok", "params": {}}) 

88 

89 return { 

90 "total_value": round(total_value, 2), 

91 "categories": categories_result, 

92 "advice": advice, 

93 }

96def compute_portfolio_health_score( 

97 categories_result: dict[str, dict], 

98 xray_entries: list[dict], 

99 xray_warn_threshold: float = XRAY_SINGLE_STOCK_WARN_PCT, 

100) -> tuple[int, str]: 

101 """ 

102 計算投資組合健康分數（0–100）及等級。 

103 

104 扣分規則： 

105 - 各分類偏移 5–10%：-8 分（每個分類獨立計算，累計扣分） 

106 - 各分類偏移 10–20%：-15 分 

107 - 各分類偏移 > 20%：-25 分 

108 - X-Ray 單一標的超過門檻（預設 15%）：每筆 -10 分（最多扣 20 分） 

109 

110 注意：每個分類的偏移懲罰會累計。例如 5 個分類均偏移 6% 時， 

111 總扣分為 -40（5 × -8），最終分數 60（caution 等級）。 

112 此設計為刻意行為：任何分類偏移均代表配置偏差，應於分數上反映。 

113 

114 等級： 

115 - healthy (80–100)：配置均衡 

116 - caution (60–79)：需要關注 

117 - alert (0–59)：需要調整 

118 

119 Args: 

120 categories_result: calculate_rebalance 回傳的 categories dict 

121 xray_entries: X-Ray 穿透式分析條目列表 

122 xray_warn_threshold: X-Ray 單一標的警戒門檻（%） 

123 

124 Returns: 

125 (score, level) tuple 

126 """ 

127 score = 100 

128 

129 for cat_data in categories_result.values(): 

130 drift = abs(cat_data.get("drift_pct", 0.0)) 

131 if drift > 20: 

132 score -= 25 

133 elif drift > 10: 

134 score -= 15 

135 elif drift > 5: 

136 score -= 8 

137 

138 xray_penalty = sum( 

139 10 

140 for entry in xray_entries 

141 if entry.get("total_weight_pct", 0.0) > xray_warn_threshold 

142 ) 

143 score -= min(xray_penalty, 20) 

144 

145 score = max(0, min(100, score)) 

146 

147 if score >= 80: 

148 level = "healthy" 

149 elif score >= 60: 

150 level = "caution" 

151 else: 

152 level = "alert" 

153 

154 return score, level
