56def format_withdrawal_telegram( 

57 plan: WithdrawalPlan, display_currency: str = "USD", lang: str = "zh-TW" 

58) -> str: 

59 """ 

60 將 WithdrawalPlan 格式化為 Telegram HTML 訊息。 

61 

62 Args: 

63 plan: domain.withdrawal.WithdrawalPlan 實例 

64 display_currency: 顯示幣別 

65 lang: 語言代碼 

66 

67 Returns: 

68 Telegram HTML 格式訊息字串 

69 """ 

70 from domain.constants import CATEGORY_ICON 

71 

72 parts: list[str] = [ 

73 t( 

74 "formatter.withdrawal_header", 

75 lang=lang, 

76 amount=f"{plan.target_amount:,.2f}", 

77 currency=display_currency, 

78 ), 

79 ] 

80 

81 if not plan.recommendations: 

82 parts.append(t("formatter.withdrawal_no_holdings", lang=lang)) 

83 return "\n".join(parts) 

84 

85 parts.append(t("formatter.withdrawal_recommendations", lang=lang)) 

86 for i, rec in enumerate(plan.recommendations, 1): 

87 icon = CATEGORY_ICON.get(rec.category, "📊") 

88 pl_text = "" 

89 if rec.unrealized_pl is not None: 

90 pl_sign = "+" if rec.unrealized_pl >= 0 else "" 

91 pl_text = t( 

92 "formatter.withdrawal_pl", 

93 lang=lang, 

94 sign=pl_sign, 

95 amount=f"{rec.unrealized_pl:,.2f}", 

96 currency=display_currency, 

97 ) 

98 priority_label = t( 

99 f"formatter.priority_{['rebalance', 'tax', 'liquidity'][rec.priority - 1]}", 

100 lang=lang, 

101 ) 

102 parts.append( 

103 f"\n{i}. {icon} <b>{rec.ticker}</b> ({rec.category})" 

104 f" — {t('formatter.sell', lang=lang)} {rec.quantity_to_sell:,.4g} " 

105 f"{t('formatter.shares', lang=lang)}" 

106 f"（{rec.sell_value:,.2f} {display_currency}）" 

107 f"\n {t('formatter.reason', lang=lang)}：{rec.reason}" 

108 f"\n {t('formatter.priority', lang=lang)}：{priority_label}" 

109 f"{pl_text}" 

110 ) 

111 

112 parts.append( 

113 t( 

114 "formatter.withdrawal_total", 

115 lang=lang, 

116 amount=f"{plan.total_sell_value:,.2f}", 

117 currency=display_currency, 

118 ) 

119 ) 

120 

121 if plan.shortfall > 0: 

122 parts.append( 

123 t( 

124 "formatter.withdrawal_shortfall", 

125 lang=lang, 

126 amount=f"{plan.shortfall:,.2f}", 

127 currency=display_currency, 

128 ) 

129 ) 

130 

131 return "\n".join(parts)

148def format_guru_filing_digest(summaries: list[dict], lang: str = "zh-TW") -> str: 

149 """ 

150 將多位大師的 13F 季報摘要格式化為 Telegram HTML 訊息。 

151 

152 Args: 

153 summaries: list of filing summary dicts（來自 filing_service，需包含 

154 guru_display_name, report_date, new_positions, sold_out, 

155 increased, decreased, top_holdings） 

156 lang: 語言代碼 

157 

158 Returns: 

159 Telegram HTML 格式字串 

160 """ 

161 if not summaries: 

162 return t("guru.digest_no_updates", lang=lang) 

163 

164 report_date = summaries[0].get("report_date", "") 

165 parts: list[str] = [ 

166 t("guru.filing_digest_title", lang=lang, report_date=report_date), 

167 "", 

168 ] 

169 

170 for summary in summaries: 

171 name = summary.get("guru_display_name", "") 

172 new_pos = summary.get("new_positions", 0) 

173 sold = summary.get("sold_out", 0) 

174 increased = summary.get("increased", 0) 

175 decreased = summary.get("decreased", 0) 

176 

177 parts.append(f"<b>{name}</b>") 

178 if new_pos or sold or increased or decreased: 

179 parts.append( 

180 t( 

181 "guru.digest_changes", 

182 lang=lang, 

183 new=new_pos, 

184 sold=sold, 

185 inc=increased, 

186 dec=decreased, 

187 ) 

188 ) 

189 else: 

190 parts.append(t("guru.digest_no_changes", lang=lang)) 

191 

192 # Top 3 holdings for brevity 

193 for h in summary.get("top_holdings", [])[:3]: 

194 icon = _HOLDING_ACTION_ICON.get(h.get("action", ""), "⚪") 

195 ticker = h.get("ticker") or h.get("cusip", "") 

196 weight = h.get("weight_pct") or 0.0 

197 parts.append(f" {icon} {ticker} ({weight:.1f}%)") 

198 parts.append("") 

199 

200 parts.append(t("guru.lagging_disclaimer_short", lang=lang)) 

201 return "\n".join(parts)

204def format_weekly_digest_html( 

205 *, 

206 lang: str, 

207 title: str, 

208 portfolio_value_line: str | None, 

209 benchmark_line: str | None, 

210 health_line: str, 

211 fear_greed_line: str, 

212 top_movers_lines: list[str], 

213 non_normal: list[dict], 

214 signal_changes: dict[str, int], 

215 drift_lines: list[str], 

216 smart_money_lines: list[str], 

217 all_normal_line: str, 

218) -> str: 

219 """ 

220 Assemble the full weekly digest Telegram HTML message from pre-built sections. 

221 

222 All section strings are already translated by the caller; this function is 

223 responsible only for ordering, grouping, and HTML structure. 

224 

225 Args: 

226 lang: language code passed to t() for section header translations 

227 title: report title line 

228 portfolio_value_line: portfolio value + WoW line, or None if no data 

229 benchmark_line: S&P 500 benchmark line, or None if unavailable 

230 health_line: health-score line 

231 fear_greed_line: fear & greed line 

232 top_movers_lines: list of individual mover lines (may be empty) 

233 non_normal: list of dicts with keys ticker, cat_label, signal 

234 signal_changes: mapping of ticker → change count for the period 

235 drift_lines: list of formatted drift lines (may be empty) 

236 smart_money_lines: list of formatted resonance-alert lines (may be empty) 

237 all_normal_line: translated "all positions normal" string 

238 

239 Returns: 

240 Telegram HTML formatted string (uses <b> tags for section headers) 

241 """ 

242 parts: list[str] = [f"<b>{title}</b>", ""] 

243 

244 # --- Portfolio value + benchmark --- 

245 if portfolio_value_line: 

246 parts.append(portfolio_value_line) 

247 if benchmark_line: 

248 parts.append(benchmark_line) 

249 if portfolio_value_line or benchmark_line: 

250 parts.append("") 

251 

252 # --- Health + Fear & Greed --- 

253 parts.append(health_line) 

254 parts.append(fear_greed_line) 

255 parts.append("") 

256 

257 # --- Top movers --- 

258 if top_movers_lines: 

259 parts.append(f"<b>{t('notification.top_movers_title', lang=lang)}</b>") 

260 parts.extend(top_movers_lines) 

261 parts.append("") 

262 

263 # --- Active signals --- 

264 if non_normal: 

265 parts.append(f"<b>{t('notification.abnormal_stocks', lang=lang)}</b>") 

266 for item in non_normal: 

267 parts.append( 

268 f" • {item['ticker']}（{item['cat_label']}）→ {item['signal']}" 

269 ) 

270 parts.append("") 

271 

272 # --- Signal changes --- 

273 if signal_changes: 

274 parts.append(f"<b>{t('notification.signal_changes', lang=lang)}</b>") 

275 change_label = t("notification.change_label", lang=lang) 

276 times_label = t("notification.times_label", lang=lang) 

277 for tk, count in sorted(signal_changes.items(), key=lambda x: -x[1]): 

278 parts.append(f" • {tk}：{change_label} {count} {times_label}") 

279 parts.append("") 

280 

281 # --- Allocation drift --- 

282 if drift_lines: 

283 parts.append(f"<b>{t('notification.drift_title', lang=lang)}</b>") 

284 parts.extend(drift_lines) 

285 parts.append("") 

286 

287 # --- Smart money --- 

288 if smart_money_lines: 

289 parts.append(f"<b>{t('notification.smart_money_title', lang=lang)}</b>") 

290 parts.extend(smart_money_lines) 

291 parts.append("") 

292 

293 # --- All normal (only when no signals and no changes) --- 

294 if not non_normal and not signal_changes: 

295 parts.append(all_normal_line) 

296 

297 # Strip trailing blank lines 

298 while parts and parts[-1] == "": 

299 parts.pop() 

300 

301 return "\n".join(parts)
