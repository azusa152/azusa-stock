56def sync_guru_filing(session: Session, guru_id: int) -> dict: 

57 """ 

58 同步指定大師的最新 13F 季報。 

59 

60 流程： 

61 1. 取得大師資料與 EDGAR 最新兩筆申報 

62 2. 若最新申報已存在資料庫則跳過（冪等） 

63 3. 委派給 _sync_single_filing() 執行實際同步 

64 

65 Args: 

66 session: Database session 

67 guru_id: Guru.id 

68 

69 Returns: 

70 dict with keys: 

71 guru_id, guru_display_name, status ("synced" | "skipped" | "error"), 

72 accession_number, report_date, filing_date, 

73 total_holdings, new_positions, sold_out, increased, decreased, 

74 top_holdings (list of top-N holdings by weight) 

75 """ 

76 guru = find_guru_by_id(session, guru_id) 

77 if guru is None: 

78 logger.warning("sync_guru_filing 找不到大師：ID=%d", guru_id) 

79 return {"guru_id": guru_id, "status": "error", "error": "guru not found"} 

80 

81 edgar_filings = get_latest_13f_filings(guru.cik, count=2) 

82 if not edgar_filings: 

83 logger.warning("EDGAR 無 13F 申報：%s (%s)", guru.display_name, guru.cik) 

84 return { 

85 "guru_id": guru_id, 

86 "guru_display_name": guru.display_name, 

87 "status": "error", 

88 "error": "no 13F filings found on EDGAR", 

89 } 

90 

91 return _sync_single_filing(session, guru, edgar_filings[0])

94def backfill_guru_filings( 

95 session: Session, 

96 guru_id: int, 

97 years: int = GURU_BACKFILL_YEARS, 

98 _today: date | None = None, 

99) -> dict: 

100 """ 

101 回填指定大師最近 N 年的 13F 歷史申報。 

102 

103 流程： 

104 1. 從 EDGAR 取回最多 GURU_BACKFILL_FILING_COUNT 筆申報 

105 2. 過濾出在 years 年窗口內的申報 

106 3. 依 report_date 升冪排序（舊→新），確保 diff 鏈正確 

107 4. 逐筆呼叫 _sync_single_filing()（冪等：已同步者自動跳過） 

108 5. 回傳摘要 dict 

109 

110 Args: 

111 session: Database session 

112 guru_id: Guru.id 

113 years: 回填年數（預設 GURU_BACKFILL_YEARS） 

114 _today: 參考日期（測試注入用；正常呼叫留 None 使用 date.today()） 

115 

116 Returns: 

117 dict with keys: guru_id, guru_display_name, total_filings, synced, skipped, errors 

118 """ 

119 guru = find_guru_by_id(session, guru_id) 

120 if guru is None: 

121 logger.warning("backfill_guru_filings 找不到大師：ID=%d", guru_id) 

122 return { 

123 "guru_id": guru_id, 

124 "status": "error", 

125 "error": "guru not found", 

126 "total_filings": 0, 

127 "synced": 0, 

128 "skipped": 0, 

129 "errors": 0, 

130 } 

131 

132 edgar_filings = get_latest_13f_filings(guru.cik, count=GURU_BACKFILL_FILING_COUNT) 

133 if not edgar_filings: 

134 logger.info("EDGAR 無 13F 申報可回填：%s (%s)", guru.display_name, guru.cik) 

135 return { 

136 "guru_id": guru_id, 

137 "guru_display_name": guru.display_name, 

138 "total_filings": 0, 

139 "synced": 0, 

140 "skipped": 0, 

141 "errors": 0, 

142 } 

143 

144 cutoff = (_today or date.today()) - timedelta(days=years * 365) 

145 in_window = [ 

146 f for f in edgar_filings if date.fromisoformat(f["report_date"]) >= cutoff 

147 ] 

148 # 升冪排序，讓舊季先同步，diff 鏈方向正確 

149 in_window.sort(key=lambda f: f["report_date"]) 

150 

151 synced = skipped = errors = 0 

152 for edgar_filing in in_window: 

153 try: 

154 result = _sync_single_filing(session, guru, edgar_filing) 

155 if result["status"] == "synced": 

156 synced += 1 

157 elif result["status"] == "skipped": 

158 skipped += 1 

159 else: 

160 errors += 1 

161 except Exception as exc: 

162 logger.warning( 

163 "回填申報失敗：%s %s, error=%s", 

164 guru.display_name, 

165 edgar_filing.get("accession_number"), 

166 exc, 

167 ) 

168 errors += 1 

169 

170 logger.info( 

171 "13F 回填完成：%s，窗口內 %d 筆，已同步 %d，跳過 %d，錯誤 %d", 

172 guru.display_name, 

173 len(in_window), 

174 synced, 

175 skipped, 

176 errors, 

177 ) 

178 return { 

179 "guru_id": guru_id, 

180 "guru_display_name": guru.display_name, 

181 "total_filings": len(in_window), 

182 "synced": synced, 

183 "skipped": skipped, 

184 "errors": errors, 

185 }

247def get_holding_changes( 

248 session: Session, guru_id: int, limit: int | None = None 

249) -> list[dict]: 

250 """ 

251 取得指定大師最新申報中有動作的持倉（action != UNCHANGED）。 

252 

253 結果依以下規則排序並限制筆數： 

254 1. 優先依 abs(change_pct) 降序（變動最大者） 

255 2. 次要依 weight_pct 降序（持倉權重最大者） 

256 3. 若指定 limit，僅回傳前 N 筆 

257 

258 Args: 

259 session: Database session 

260 guru_id: Guru.id 

261 limit: 最多回傳筆數（None 表示不限制） 

262 

263 Returns: 

264 持倉變動列表，每筆含 report_date, filing_date, ticker, company_name, 

265 action, change_pct, value, shares, weight_pct 

266 """ 

267 filing = find_latest_filing_by_guru(session, guru_id) 

268 if filing is None: 

269 return [] 

270 

271 holdings = find_holdings_by_filing(session, filing.id) 

272 changes = [h for h in holdings if h.action != HoldingAction.UNCHANGED.value] 

273 

274 # Sort by significance: abs(change_pct) DESC, then weight_pct DESC 

275 changes_sorted = sorted( 

276 changes, 

277 key=lambda h: ( 

278 abs(h.change_pct) if h.change_pct is not None else 0.0, 

279 h.weight_pct if h.weight_pct is not None else 0.0, 

280 ), 

281 reverse=True, 

282 ) 

283 

284 # Apply limit if specified 

285 if limit is not None and limit > 0: 

286 changes_sorted = changes_sorted[:limit] 

287 

288 return [_holding_to_dict(h, filing) for h in changes_sorted]

358def _sync_single_filing(session: Session, guru: Guru, edgar_filing_dict: dict) -> dict: 

359 """ 

360 同步單筆 EDGAR 申報至資料庫（由 sync_guru_filing 與 backfill_guru_filings 共用）。 

361 

362 流程： 

363 1. 冪等檢查（accession_number 已存在則回傳 skipped） 

364 2. 下載並解析 13F XML 

365 3. 載入前季持倉快照（在 save_filing 之前呼叫，確保取得舊季資料） 

366 4. 組裝 GuruFiling + GuruHolding，儲存至 DB 

367 5. 回傳摘要 dict 

368 

369 Args: 

370 session: Database session 

371 guru: Guru entity 

372 edgar_filing_dict: 單筆 EDGAR 申報 dict（含 accession_number, report_date 等） 

373 

374 Returns: 

375 dict with status "synced" | "skipped" | "error" 

376 """ 

377 accession_number = edgar_filing_dict["accession_number"] 

378 

379 # 冪等檢查 

380 if find_filing_by_accession(session, accession_number) is not None: 

381 logger.info( 

382 "13F 申報已存在，跳過同步：%s %s", guru.display_name, accession_number 

383 ) 

384 return { 

385 "guru_id": guru.id, 

386 "guru_display_name": guru.display_name, 

387 "status": "skipped", 

388 "accession_number": accession_number, 

389 "report_date": edgar_filing_dict["report_date"], 

390 "filing_date": edgar_filing_dict["filing_date"], 

391 } 

392 

393 # 下載並解析持倉明細 

394 raw_holdings = fetch_13f_filing_detail(accession_number, guru.cik) 

395 if not raw_holdings: 

396 logger.warning( 

397 "13F 持倉明細解析失敗：%s %s", guru.display_name, accession_number 

398 ) 

399 return { 

400 "guru_id": guru.id, 

401 "guru_display_name": guru.display_name, 

402 "status": "error", 

403 "error": "failed to fetch or parse infotable XML", 

404 } 

405 

406 # 載入前季持倉快照（必須在 save_filing 之前呼叫，此時 DB 中最新申報即為前季） 

407 prev_holdings_map = _snapshot_prev_holdings(session, guru.id) 

408 

409 # 計算總持倉市值（用於 weight_pct） 

410 total_value = sum(h["value"] for h in raw_holdings) 

411 

412 # 組裝 GuruFiling 

413 filing = save_filing( 

414 session, 

415 GuruFiling( 

416 guru_id=guru.id, 

417 accession_number=accession_number, 

418 report_date=edgar_filing_dict["report_date"], 

419 filing_date=edgar_filing_dict["filing_date"], 

420 total_value=total_value, 

421 holdings_count=len(raw_holdings), 

422 filing_url=edgar_filing_dict.get("filing_url", ""), 

423 ), 

424 ) 

425 

426 # 組裝並分類 GuruHolding 列表（含前季完全消失的 SOLD_OUT） 

427 holdings = _build_holdings( 

428 raw_holdings, filing, guru, prev_holdings_map, total_value 

429 ) 

430 save_holdings_batch(session, holdings) 

431 

432 summary = _build_summary(guru, filing, holdings) 

433 logger.info( 

434 "13F 同步完成：%s %s，持倉 %d 筆，新建倉 %d，清倉 %d", 

435 guru.display_name, 

436 accession_number, 

437 len(holdings), 

438 summary["new_positions"], 

439 summary["sold_out"], 

440 ) 

441 return summary

444def _snapshot_prev_holdings(session: Session, guru_id: int) -> dict[str, float]: 

445 """ 

446 載入資料庫中現有最新申報的 cusip → shares 快照，作為本次 diff 的基準。 

447 

448 重要：此函式必須在 save_filing（新申報）之前呼叫，否則會取到剛存入的 

449 本季資料而非前季資料。 

450 

451 Args: 

452 session: Database session 

453 guru_id: Guru.id 

454 

455 Returns: 

456 cusip → shares 映射；若無前季申報則回傳空 dict 

457 """ 

458 prev_filing = find_latest_filing_by_guru(session, guru_id) 

459 if prev_filing is None: 

460 return {} 

461 

462 prev_holdings = find_holdings_by_filing(session, prev_filing.id) 

463 return {h.cusip: h.shares for h in prev_holdings}

466def _build_holdings( 

467 raw_holdings: list[dict], 

468 filing: GuruFiling, 

469 guru: Guru, 

470 prev_map: dict[str, float], 

471 total_value: float, 

472) -> list[GuruHolding]: 

473 """ 

474 將 EDGAR 原始持倉列表轉換為 GuruHolding 物件，含分類與計算。 

475 

476 同時處理前季有但本季完全消失的 CUSIP（SOLD_OUT）， 

477 這類倉位不出現在 EDGAR XML 中，需從 prev_map 補充建立。 

478 """ 

479 # Pass 1 — 解析所有 CUSIP → ticker 映射（含本季 + 前季清倉） 

480 current_cusips: set[str] = set() 

481 cusip_to_ticker: dict[str, str | None] = {} 

482 

483 for raw in raw_holdings: 

484 cusip = raw["cusip"] 

485 current_cusips.add(cusip) 

486 cusip_to_ticker[cusip] = map_cusip_to_ticker(cusip, raw["company_name"]) 

487 

488 for cusip in prev_map: 

489 if cusip not in current_cusips: 

490 cusip_to_ticker[cusip] = map_cusip_to_ticker(cusip, "") 

491 

492 # Pass 2 — 批次解析 sector（每個唯一 ticker 只呼叫一次 get_ticker_sector） 

493 unique_tickers = {t for t in cusip_to_ticker.values() if t} 

494 ticker_to_sector: dict[str, str | None] = { 

495 t: get_ticker_sector(t) for t in unique_tickers 

496 } 

497 

498 holdings = [] 

499 

500 for raw in raw_holdings: 

501 cusip = raw["cusip"] 

502 current_shares = raw["shares"] 

503 previous_shares = prev_map.get(cusip) 

504 

505 action = classify_holding_change(current_shares, previous_shares) 

506 change_pct = ( 

507 compute_change_pct(current_shares, previous_shares) 

508 if previous_shares is not None 

509 else None 

510 ) 

511 weight_pct = compute_holding_weight(raw["value"], total_value) 

512 ticker = cusip_to_ticker[cusip] 

513 sector = ticker_to_sector.get(ticker) if ticker else None 

514 

515 holdings.append( 

516 GuruHolding( 

517 filing_id=filing.id, 

518 guru_id=guru.id, 

519 cusip=cusip, 

520 ticker=ticker, 

521 company_name=raw["company_name"], 

522 value=raw["value"], 

523 shares=current_shares, 

524 action=action.value, 

525 change_pct=change_pct, 

526 weight_pct=weight_pct, 

527 sector=sector, 

528 ) 

529 ) 

530 

531 # 前季存在但本季完全消失 → SOLD_OUT 

532 for cusip, _prev_shares in prev_map.items(): 

533 if cusip not in current_cusips: 

534 ticker = cusip_to_ticker[cusip] 

535 sector = ticker_to_sector.get(ticker) if ticker else None 

536 holdings.append( 

537 GuruHolding( 

538 filing_id=filing.id, 

539 guru_id=guru.id, 

540 cusip=cusip, 

541 ticker=ticker, 

542 company_name="", 

543 value=0.0, 

544 shares=0.0, 

545 action=HoldingAction.SOLD_OUT.value, 

546 change_pct=-100.0, 

547 weight_pct=0.0, 

548 sector=sector, 

549 ) 

550 ) 

551 

552 return holdings

555def _build_summary(guru: Guru, filing: GuruFiling, holdings: list[GuruHolding]) -> dict: 

556 """組裝 sync/summary 回傳 dict。""" 

557 action_counts: dict[str, int] = { 

558 HoldingAction.NEW_POSITION.value: 0, 

559 HoldingAction.SOLD_OUT.value: 0, 

560 HoldingAction.INCREASED.value: 0, 

561 HoldingAction.DECREASED.value: 0, 

562 } 

563 for h in holdings: 

564 if h.action in action_counts: 

565 action_counts[h.action] += 1 

566 

567 top_holdings = sorted( 

568 [_holding_to_dict(h, filing) for h in holdings if h.weight_pct is not None], 

569 key=lambda x: x["weight_pct"] or 0, 

570 reverse=True, 

571 )[:GURU_TOP_HOLDINGS_COUNT] 

572 

573 return { 

574 "guru_id": guru.id, 

575 "guru_display_name": guru.display_name, 

576 "status": "synced", 

577 "accession_number": filing.accession_number,
