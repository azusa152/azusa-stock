164class UserPreferences(SQLModel, table=True): 

165 """使用者偏好設定（跨裝置同步）。""" 

166 

167 user_id: str = Field( 

168 default=DEFAULT_USER_ID, primary_key=True, description="使用者 ID" 

169 ) 

170 language: str = Field(default=DEFAULT_LANGUAGE, description="偏好語言") 

171 privacy_mode: bool = Field(default=False, description="是否啟用隱私模式") 

172 notification_preferences: str = Field( 

173 default=_json.dumps(DEFAULT_NOTIFICATION_PREFERENCES), 

174 sa_column=Column(String, default=_json.dumps(DEFAULT_NOTIFICATION_PREFERENCES)), 

175 description="通知偏好 JSON（各類通知的啟用/停用）", 

176 ) 

177 

178 def get_notification_prefs(self) -> dict[str, bool]: 

179 """解析通知偏好 JSON，缺少的 key 以預設值填補。""" 

180 try: 

181 stored = _json.loads(self.notification_preferences) 

182 except (TypeError, _json.JSONDecodeError): 

183 stored = {} 

184 return {**DEFAULT_NOTIFICATION_PREFERENCES, **stored} 

185 

186 def set_notification_prefs(self, prefs: dict[str, bool]) -> None: 

187 """合併並序列化通知偏好。""" 

188 merged = {**DEFAULT_NOTIFICATION_PREFERENCES, **prefs} 

189 self.notification_preferences = _json.dumps(merged)
