68class _EdgarRateLimiter: 

69 """Thread-safe rate limiter for SEC EDGAR (10 req/sec).""" 

70 

71 def __init__(self, calls_per_second: float = SEC_EDGAR_RATE_LIMIT_CPS) -> None: 

72 self._min_interval = 1.0 / calls_per_second 

73 self._lock = threading.Lock() 

74 self._last_call = 0.0 

75 

76 def wait(self) -> None: 

77 with self._lock: 

78 now = time.monotonic() 

79 elapsed = now - self._last_call 

80 if elapsed < self._min_interval: 

81 time.sleep(self._min_interval - elapsed) 

82 self._last_call = time.monotonic() 

83 

84

141def _discover_infotable_filename(accession_path: str, cik: str) -> str | None: 

142 """ 

143 從 EDGAR filing index.json 動態探索 information table XML 檔名。 

144 

145 EDGAR 13F filings 的 information table XML 檔名不固定，可能是： 

146 - infotable.xml 

147 - 50240.xml (Berkshire 使用的實際檔名) 

148 - 13f_infotable.xml 

149 - 其他數字檔名 

150 

151 策略：取得 index.json，找出所有 .xml 檔案，排除 primary_doc.xml， 

152 回傳第一個符合的檔名。 

153 

154 Args: 

155 accession_path: Accession number without hyphens (e.g. "000119312526054580"). 

156 cik: 10-digit zero-padded CIK (or stripped CIK will be handled). 

157 

158 Returns: 

159 Discovered XML filename (without path), or None on any error. 

160 """ 

161 index_url = ( 

162 f"{SEC_EDGAR_ARCHIVES_BASE_URL}/Archives/edgar/data/" 

163 f"{cik.lstrip('0')}/{accession_path}/index.json" 

164 ) 

165 

166 try: 

167 index_data = _http_get_json(index_url) 

168 items = index_data.get("directory", {}).get("item", []) 

169 

170 # Find all .xml files that are NOT primary_doc.xml 

171 for item in items: 

172 name = item.get("name", "") 

173 if name.endswith(".xml") and name != "primary_doc.xml": 

174 logger.debug( 

175 "EDGAR infotable 檔名探索成功：%s → %s", accession_path, name 

176 ) 

177 return name 

178 

179 logger.debug("EDGAR index 中找不到 infotable XML：%s", accession_path) 

180 return None 

181 

182 except Exception as exc: 

183 logger.debug("EDGAR infotable 檔名探索失敗：%s, error=%s", accession_path, exc) 

184 return None

192def fetch_company_filings(cik: str) -> dict: 

193 """ 

194 取得 EDGAR 公司申報索引 (submissions JSON)。 

195 結果以 CIK 為 key，L1+L2 雙層快取。 

196 

197 Args: 

198 cik: 10-digit zero-padded SEC CIK code. 

199 

200 Returns: 

201 EDGAR submissions JSON dict，失敗時回傳 {"error": str}。 

202 """ 

203 cached = _filing_cache.get(cik) 

204 if cached is not None: 

205 logger.debug("EDGAR submissions L1 命中：CIK=%s", cik) 

206 return cached 

207 

208 disk_key = f"{DISK_KEY_GURU_FILING}:submissions:{cik}" 

209 disk_cached = _disk_get(disk_key) 

210 if disk_cached is not None: 

211 logger.debug("EDGAR submissions L2 命中：CIK=%s", cik) 

212 _filing_cache[cik] = disk_cached 

213 return disk_cached 

214 

215 url = f"{SEC_EDGAR_BASE_URL}/submissions/CIK{cik}.json" 

216 try: 

217 result = _http_get_json(url) 

218 _filing_cache[cik] = result 

219 _disk_set(disk_key, result, DISK_GURU_FILING_TTL) 

220 logger.debug("EDGAR submissions 已取得並快取：CIK=%s", cik) 

221 return result 

222 except Exception as exc: 

223 logger.warning("EDGAR submissions 取得失敗：CIK=%s, error=%s", cik, exc) 

224 return {"error": str(exc)}

227def get_latest_13f_filings(cik: str, count: int = 2) -> list[dict]: 

228 """ 

229 從 EDGAR submissions 索引中篩選最新 N 筆 13F-HR 申報。 

230 

231 Args: 

232 cik: 10-digit zero-padded SEC CIK code. 

233 count: 回傳筆數（預設 2，用於季度比對）。 

234 

235 Returns: 

236 list of dicts with keys: accession_number, filing_date, report_date, 

237 primary_doc, filing_url. Empty list on failure. 

238 """ 

239 submissions = fetch_company_filings(cik) 

240 if "error" in submissions: 

241 return [] 

242 

243 try: 

244 filings = submissions.get("filings", {}).get("recent", {}) 

245 forms = filings.get("form", []) 

246 accessions = filings.get("accessionNumber", []) 

247 filing_dates = filings.get("filingDate", []) 

248 report_dates = filings.get("reportDate", []) 

249 primary_docs = filings.get("primaryDocument", []) 

250 

251 results = [] 

252 for i, form in enumerate(forms): 

253 if form == "13F-HR": 

254 accession = accessions[i] 

255 accession_path = accession.replace("-", "") 

256 filing_url = ( 

257 f"{SEC_EDGAR_ARCHIVES_BASE_URL}/Archives/edgar/data/" 

258 f"{cik.lstrip('0')}/{accession_path}/{primary_docs[i]}" 

259 ) 

260 results.append( 

261 { 

262 "accession_number": accession, 

263 "accession_path": accession_path, 

264 "filing_date": filing_dates[i], 

265 "report_date": report_dates[i], 

266 "primary_doc": primary_docs[i], 

267 "filing_url": filing_url, 

268 } 

269 ) 

270 if len(results) >= count: 

271 break 

272 

273 logger.debug("EDGAR 13F-HR 申報索引取得：CIK=%s, 共 %d 筆", cik, len(results)) 

274 return results 

275 

276 except Exception as exc: 

277 logger.warning("EDGAR 13F 索引解析失敗：CIK=%s, error=%s", cik, exc) 

278 return []

281def fetch_13f_filing_detail(accession_number: str, cik: str) -> list[dict]: 

282 """ 

283 下載並解析 13F information table XML，回傳持倉明細列表。 

284 

285 Args: 

286 accession_number: SEC accession number (e.g. "0001067983-24-000006"). 

287 cik: 10-digit zero-padded CIK (used to build the EDGAR archive URL). 

288 

289 Returns: 

290 list of dicts with keys: cusip, company_name, value, shares. 

291 Empty list on parse failure or network error. 

292 """ 

293 accession_path = accession_number.replace("-", "") 

294 

295 # Discover the actual XML filename from the filing index 

296 xml_filename = _discover_infotable_filename(accession_path, cik) 

297 if xml_filename is None: 

298 # Fallback to the common default filename 

299 xml_filename = "infotable.xml" 

300 logger.debug( 

301 "EDGAR infotable 檔名探索失敗，使用預設檔名：%s → %s", 

302 accession_number, 

303 xml_filename, 

304 ) 

305 

306 # Filing documents are served by www.sec.gov, not data.sec.gov 

307 base = f"{SEC_EDGAR_ARCHIVES_BASE_URL}/Archives/edgar/data/{cik.lstrip('0')}/{accession_path}" 

308 xml_url = f"{base}/{xml_filename}" 

309 

310 # Include filename in cache key to prevent stale cache when filename changes 

311 cache_key = f"{DISK_KEY_GURU_FILING}:infotable:{accession_number}:{xml_filename}" 

312 disk_cached = _disk_get(cache_key) 

313 if disk_cached is not None: 

314 logger.debug("EDGAR infotable L2 命中：%s", accession_number) 

315 return disk_cached 

316 

317 try: 

318 xml_text = _http_get_text(xml_url) 

319 holdings = _parse_13f_xml(xml_text) 

320 _disk_set(cache_key, holdings, DISK_GURU_FILING_TTL) 

321 logger.info( 

322 "EDGAR infotable 解析完成：%s (%s), %d 筆持倉", 

323 accession_number, 

324 xml_filename, 

325 len(holdings), 

326 ) 

327 return holdings 

328 except Exception as exc: 

329 logger.warning("EDGAR infotable 取得失敗：%s, error=%s", accession_number, exc) 

330 return []

333def map_cusip_to_ticker(cusip: str, company_name: str) -> str | None: 

334 """ 

335 盡力將 CUSIP 轉換為股票代號。 

336 

337 策略（依優先順序）： 

338 1. 靜態本地查找表 (_CUSIP_MAP) 

339 2. 由公司名稱推斷常見代號（有限） 

340 3. 回傳 None（未來 Phase 可串接 EDGAR company search API） 

341 

342 Args: 

343 cusip: 9-character CUSIP identifier. 

344 company_name: Company name from 13F filing (best-effort hint). 

345 

346 Returns: 

347 Ticker symbol string or None if unmappable. 

348 """ 

349 # Normalize 

350 cusip = cusip.strip().upper() 

351 name_upper = company_name.strip().upper() 

352 

353 # 1. Static lookup table (most reliable, covers top holdings) 

354 ticker = _CUSIP_MAP.get(cusip) 

355 if ticker: 

356 return ticker 

357 

358 # 2. Name-based heuristic for well-known names 

359 for fragment, sym in _NAME_HINTS.items(): 

360 if fragment in name_upper: 

361 return sym 

362 

363 return None

371def _parse_13f_xml(xml_text: str) -> list[dict]: 

372 """ 

373 解析 SEC 13F information table XML，回傳持倉列表。 

374 

375 EDGAR 13F XML 使用兩種命名空間變體： 

376 - 新格式 (2013+): ns 'com/xbrl/dim/2011' 

377 - 舊格式: no namespace 

378 

379 Returns: 

380 list of dicts: cusip, company_name, value (thousands USD), shares. 

381 """ 

382 # Strip namespace declarations to simplify parsing 

383 xml_clean = xml_text 

384 for ns_prefix in ( 

385 'xmlns="', 

386 "xmlns:ns1=", 

387 "xmlns:xsi=", 

388 " xsi:schemaLocation=", 

389 ): 

390 if ns_prefix in xml_clean: 

391 # Remove namespace URIs but keep tag names 

392 xml_clean = re.sub(r'\s*xmlns[^"]*"[^"]*"', "", xml_clean) 

393 xml_clean = re.sub(r'\s*xsi:schemaLocation="[^"]*"', "", xml_clean) 

394 break 

395 

396 # Strip any residual namespace prefixes from tag names (ns1:infoTable → infoTable) 

397 xml_clean = re.sub(r"<(/?)[\w]+:", r"<\1", xml_clean) 

398 

399 try: 

400 root = ET.fromstring(xml_clean) 

401 except ET.ParseError as exc: 

402 logger.warning("13F XML 解析失敗：%s", exc) 

403 return [] 

404 

405 holdings = [] 

406 # Both wrapper tag names used across EDGAR history 

407 for entry in root.iter("infoTable"): 

408 try: 

409 cusip = _xml_text(entry, "cusip") 

410 company_name = _xml_text(entry, "nameOfIssuer") 

411 # value is in thousands USD 

412 value_str = _xml_text(entry, "value") 

413 # shrsOrPrnAmt contains sshPrnamt (shares) and sshPrnamtType 

414 sh_node = entry.find("shrsOrPrnAmt") 

415 shares_str = _xml_text(sh_node, "sshPrnamt") if sh_node is not None else "0" 

416 

417 if not cusip or not company_name: 

418 continue 

419 

420 holdings.append( 

421 { 

422 "cusip": cusip.strip().upper(), 

423 "company_name": company_name.strip(), 

424 "value": float(value_str or 0), 

425 "shares": float(shares_str or 0), 

426 } 

427 ) 

428 except Exception as exc: 

429 logger.debug("13F infoTable 條目解析跳過：%s", exc) 

430 continue 

431 

432 return holdings

435def _xml_text(node, tag: str) -> str: 

436 """Safely extract text from an XML child element.""" 

437 if node is None: 

438 return "" 

439 child = node.find(tag) 

440 if child is None or child.text is None: 

441 return "" 

442 return child.text.strip()

445def _date_to_quarter(date_str: str) -> int: 

446 """Convert YYYY-MM-DD to quarter number (1–4).""" 

447 month = int(date_str[5:7]) 

448 return (month - 1) // 3 + 1
